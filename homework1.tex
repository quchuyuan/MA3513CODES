\begin{homeworkProblem}
   

    \textbf{Solution}\\
$H_0:\mu=38.5$ and $H_a:\mu\neq38.5$\\
$\bar{X}=\frac{1}{15}\sum_{n=1}^{15}X_n=40.787$\\
$S=\sqrt{\frac{\sum_{i=1}^{15}(x_i-\bar{x})^2}{15-1}}=3.37$\\
$t=\frac{\bar{X}-\mu}{S/\sqrt{n}}=\frac{40.787-38.5}{3.37/\sqrt{15}}=2.63$\\
DF=n-1=14\\
$p=0.0198<\alpha$\\
Therefore, we can reject Null Hypothesis and  we  conclude  that  it  is  reasonable  to  suppose the current mean retail price for oranges is different from the mean of 38.5 cents per pound in Year 2014.\\
    
\end{homeworkProblem}

   DF=6+6-2=10\\

\end{homeworkProblem}

\begin{homeworkProblem}
   4.\textbf{Solution:}\\
   
   (a)\\
   $H_0:\sigma\ge2.4$\\
   $\sigma^2=\frac{\sum(X-\mu)^2}{N}=\frac{\sum(X-\frac{\sum X}{N})^2}{N}=\frac{\sum X^2}{N}-(\frac{\sum X}{N})^2=396.84-392.04=4.8$\\
   $SD=\sqrt{4.8}=2.19$\\
   $T=\frac{(N-1)s^2}{\sigma^2}=\frac{24\times 4.8}{5.7}=20.21$\\
   $\chi^2_{(0.95,N-1)}=36.42$\\
   $T<\chi^2_{(0.95,N-1)}$\\
   Therefore, we fail to reject the hypothesis and  we  conclude  that  it  is  reasonable  to  suppose standard deviation of the amount of carbon monoxide emitted from Model-A passenger cars is less than 2.4 grams per mile.\\
   
   (b)\\
   $H_0:\mu_1-\mu_2\le 2$ and $H_a:\mu_1-\mu_2>2$\\
   $\overline{X_1}=19.8$ $S_1=2.19$\\
   $\overline{X_2}=16$ $S_2=3.33$\\
   $T=\frac{\overline{X_1}-\overline{X_2}-2}{\sqrt{\frac{S_1^2}{n_1}+\frac{S_2^2}{n_2}}}=\frac{19.8-16-2}{\sqrt{\frac{2.19^2}{25}+\frac{3.33^2}{13}}}=1.76$\\
   $DF=\frac{[\frac{S_1^2}{n_1}+\frac{S_2^2}{n_2}]^2}{\frac{(S_1^2/n_1)^2}{n_1-1}+\frac{(S_2^2/n_2)^2}{n_2-1}}=\frac{[\frac{2.19^2}{25}+\frac{3.33^2}{13}]^2}{\frac{(2.19^2/25)^2}{25-1}+\frac{(3.33^2/13)^2}{13-1}}=17.56$\\
   $p=0.048<\alpha$\\
   Therefore,  we have sufficient evidnece to rejet Null Hypothesis.\\
   
   (c)\\
   $H_0:\sigma_1=\sigma_2$ and $H_a:\sigma_1 \ne \sigma_2$\\
   $F=\frac{s_1^2}{s_2^2}=\frac{4.8}{11.09}=0.43$\\
   $R=\{F:F>F_0.05(24,12)=2.51 or F< F_0.55(24,12)=\frac{1}{2.51}=0.398\}$\\
   $0.43\notin R$\\
   $H_0$ is  not  rejected  and  we  conclude  that  it  is  reasonable  to  suppose  the  two populations have the same variance.\\
\end{homeworkProblem}
